    <script>
        // ==========================================
        // TEACHER SERVER (LocalStorage-based sync)
        // ==========================================
        
        const TEACHER_SERVER = {
            getGameState: () => {
                return JSON.parse(localStorage.getItem('kodat_izhoda_game_state') || '{}');
            },
            
            setGameState: (state) => {
                localStorage.setItem('kodat_izhoda_game_state', JSON.stringify(state));
            },
            
            getPlayers: () => {
                return JSON.parse(localStorage.getItem('kodat_izhoda_players') || '[]');
            },
            
            generateCode: () => {
                return Math.floor(1000 + Math.random() * 9000).toString();
            }
        };

        let updateInterval = null;
        let timerInterval = null;
        let endTime = null;

        // ==========================================
        // INITIALIZATION
        // ==========================================
        
        function init() {
            const state = TEACHER_SERVER.getGameState();
            
            if (!state.classCode) {
                generateNewCode();
            } else {
                document.getElementById('class-code').textContent = state.classCode;
            }
            
            if (state.status === 'active' && state.endTime) {
                endTime = state.endTime;
                startTimer();
            }
            
            startUpdateLoop();
        }

        function generateNewCode() {
            const code = TEACHER_SERVER.generateCode();
            const state = TEACHER_SERVER.getGameState();
            state.classCode = code;
            state.status = 'waiting';
            TEACHER_SERVER.setGameState(state);
            document.getElementById('class-code').textContent = code;
        }

        // ==========================================
        // GAME CONTROL
        // ==========================================
        
        function startGame() {
            const state = TEACHER_SERVER.getGameState();
            state.status = 'active';
            state.startTime = Date.now();
            endTime = Date.now() + (45 * 60 * 1000); // 45 minutes
            state.endTime = endTime;
            TEACHER_SERVER.setGameState(state);
            
            startTimer();
            
            // Clear finished status for all players
            const players = TEACHER_SERVER.getPlayers();
            players.forEach(p => {
                p.finished = false;
                p.score = 0;
                p.level = 1;
                p.task = 1;
            });
            localStorage.setItem('kodat_izhoda_players', JSON.stringify(players));
        }

        function stopGame() {
            const state = TEACHER_SERVER.getGameState();
            state.status = 'stopped';
            state.endTime = Date.now();
            TEACHER_SERVER.setGameState(state);
            
            if (timerInterval) {
                clearInterval(timerInterval);
                timerInterval = null;
            }
            
            document.getElementById('timer-display').textContent = 'СТОП';
            document.getElementById('timer-display').style.color = '#ff003c';
        }

        function resetGame() {
            if (!confirm('Сигурен ли си? Това ще изтрие всички текущи игри и ще генерира нов код.')) return;
            
            if (timerInterval) {
                clearInterval(timerInterval);
                timerInterval = null;
            }
            
            TEACHER_SERVER.setGameState({});
            localStorage.removeItem('kodat_izhoda_players');
            
            document.getElementById('timer-display').textContent = '45:00';
            document.getElementById('timer-display').style.color = '#0aff0a';
            
            generateNewCode();
        }

        // ==========================================
        // TIMER
        // ==========================================
        
        function startTimer() {
            if (timerInterval) clearInterval(timerInterval);
            
            updateTimer();
            timerInterval = setInterval(updateTimer, 1000);
        }

        function updateTimer() {
            const now = Date.now();
            const remaining = Math.max(0, endTime - now);
            
            const m = Math.floor(remaining / 60000);
            const s = Math.floor((remaining % 60000) / 1000);
            
            document.getElementById('timer-display').textContent = 
                `${m}:${s.toString().padStart(2, '0')}`;
            
            if (remaining <= 0) {
                document.getElementById('timer-display').textContent = '00:00';
                document.getElementById('timer-display').style.color = '#ff003c';
            } else if (remaining < 5 * 60 * 1000) {
                document.getElementById('timer-display').style.color = '#ffee00';
            } else {
                document.getElementById('timer-display').style.color = '#0aff0a';
            }
        }

        // ==========================================
        // LIVE UPDATES
        // ==========================================
        
        function startUpdateLoop() {
            updatePlayers();
            updateInterval = setInterval(updatePlayers, 2000);
        }

        function updatePlayers() {
            const players = TEACHER_SERVER.getPlayers();
            const state = TEACHER_SERVER.getGameState();
            
            // Update stats
            const total = players.length;
            const active = players.filter(p => !p.finished && state.status === 'active').length;
            const finished = players.filter(p => p.finished).length;
            const avgScore = total > 0 ? Math.round(players.reduce((a, b) => a + (b.score || 0), 0) / total) : 0;
            
            document.getElementById('total-players').textContent = total;
            document.getElementById('active-players').textContent = active;
            document.getElementById('finished-players').textContent = finished;
            document.getElementById('avg-score').textContent = avgScore;
            
            // Update table
            const tbody = document.getElementById('players-tbody');
            
            if (players.length === 0) {
                tbody.innerHTML = `
                    <tr class="empty-state">
                        <td colspan="6">
                            <div class="empty-icon">👻</div>
                            <div>Все още няма играчи...</div>
                            <div style="font-size: 0.9rem; margin-top: 10px;">Сподели кода с класа!</div>
                        </td>
                    </tr>
                `;
                return;
            }
            
            // Sort by score descending
            const sorted = [...players].sort((a, b) => (b.score || 0) - (a.score || 0));
            
            tbody.innerHTML = sorted.map((p, i) => {
                const progress = ((p.level - 1) * 3 + (p.task - 1)) / 9 * 100;
                const status = p.finished ? 'finished' : (state.status === 'active' ? 'playing' : 'waiting');
                const statusText = { waiting: 'Чака', playing: 'Играе', finished: 'Готов' }[status];
                
                return `
                    <tr>
                        <td class="player-rank">${i + 1}</td>
                        <td>
                            <div class="player-name">${p.name || 'Без име'}</div>
                            <div class="player-class">${p.class || ''}</div>
                        </td>
                        <td>Н${p.level || 1} З${p.task || 1}</td>
                        <td class="player-score">${p.score || 0}</td>
                        <td>
                            <div class="player-progress">
                                <div class="progress-bar">
                                    <div class="progress-fill" style="width: ${progress}%"></div>
                                </div>
                                <span>${Math.round(progress)}%</span>
                            </div>
                        </td>
                        <td><span class="status-badge status-${status}">${statusText}</span></td>
                    </tr>
                `;
            }).join('');
        }

        // Init
        init();
    </script>

        const TEACHER_PASSWORD = '1111'; // Парола за достъп

        // TEACHER SERVER (LocalStorage-based sync)

        const TEACHER_SERVER = {
            getGameState: () => {
                return JSON.parse(localStorage.getItem('kodat_izhoda_game_state') || '{}');
            },
            
            setGameState: (state) => {
                localStorage.setItem('kodat_izhoda_game_state', JSON.stringify(state));
            },
            
            getPlayers: () => {
                return JSON.parse(localStorage.getItem('kodat_izhoda_players') || '[]');
            },
            
            generateCode: () => {
                return Math.floor(1000 + Math.random() * 9000).toString();
            }
        };

        let updateInterval = null;
        let timerInterval = null;
        let endTime = null;

        // ==========================================
        // LOGIN SYSTEM

        function checkTeacherCode() {
            const input = document.getElementById('teacher-code').value;
            
            if (input === TEACHER_PASSWORD) {
                document.getElementById('login-screen').classList.add('hidden');
                document.getElementById('main-panel').classList.remove('hidden');
                initPanel();
            } else {
                const error = document.getElementById('login-error');
                error.style.display = 'block';
                document.getElementById('teacher-code').value = '';
                document.getElementById('teacher-code').focus();
                
                // Shake animation effect
                const box = document.querySelector('.login-box');
                box.style.animation = 'shake 0.5s';
                setTimeout(() => box.style.animation = '', 500);
            }
        }

        // Enter key support
        document.getElementById('teacher-code').addEventListener('keypress', (e) => {
            if (e.key === 'Enter') checkTeacherCode();

        // Add shake animation
        const style = document.createElement('style');
        style.textContent = `

        // ==========================================
        // MAIN PANEL INITIALIZATION
        // ==========================================
        
        function initPanel() {
            const state = TEACHER_SERVER.getGameState();
            
            if (!state.classCode) {
                generateNewCode();
            } else {
                document.getElementById('class-code').textContent = state.classCode;
            }
            
            if (state.status === 'active' && state.endTime) {
                endTime = state.endTime;
                startTimer();
            }
            
            startUpdateLoop();
        }

        function generateNewCode() {
            const code = TEACHER_SERVER.generateCode();
            const state = TEACHER_SERVER.getGameState();
            state.classCode = code;
            state.status = 'waiting';
            TEACHER_SERVER.setGameState(state);
            document.getElementById('class-code').textContent = code;
        }

        // ==========================================
        // GAME CONTROL
        // ==========================================
        
        function startGame() {
            const state = TEACHER_SERVER.getGameState();
            state.status = 'active';
            state.startTime = Date.now();
            endTime = Date.now() + (45 * 60 * 1000);
            state.endTime = endTime;
            TEACHER_SERVER.setGameState(state);
            
            startTimer();
            
            const players = TEACHER_SERVER.getPlayers();
            players.forEach(p => {
                p.finished = false;
                p.score = 0;
                p.level = 1;
                p.task = 1;
            });
            localStorage.setItem('kodat_izhoda_players', JSON.stringify(players));
        }

        function stopGame() {
            const state = TEACHER_SERVER.getGameState();
            state.status = 'stopped';
            state.endTime = Date.now();
            TEACHER_SERVER.setGameState(state);
            
            if (timerInterval) {
                clearInterval(timerInterval);
                timerInterval = null;
            }
            
            document.getElementById('timer-display').textContent = 'СТОП';
            document.getElementById('timer-display').style.color = '#ff003c';
        }

        function resetGame() {
            if (!confirm('Сигурен ли си? Това ще изтрие всички текущи игри и ще генерира нов код.')) return;
            
            if (timerInterval) {
                clearInterval(timerInterval);
                timerInterval = null;
            }
            
            TEACHER_SERVER.setGameState({});
            localStorage.removeItem('kodat_izhoda_players');
            
            document.getElementById('timer-display').textContent = '45:00';
            document.getElementById('timer-display').style.color = '#0aff0a';
            
            generateNewCode();
        }

        // ==========================================
        // TIMER
        // ==========================================
        
        function startTimer() {
            if (timerInterval) clearInterval(timerInterval);
            updateTimer();
            timerInterval = setInterval(updateTimer, 1000);
        }

        function updateTimer() {
            const now = Date.now();
            const remaining = Math.max(0, endTime - now);
            
            const m = Math.floor(remaining / 60000);
            const s = Math.floor((remaining % 60000) / 1000);
            
            document.getElementById('timer-display').textContent = 
                `${m}:${s.toString().padStart(2, '0')}`;
            
            if (remaining <= 0) {
                document.getElementById('timer-display').textContent = '00:00';
                document.getElementById('timer-display').style.color = '#ff003c';
            } else if (remaining < 5 * 60 * 1000) {
                document.getElementById('timer-display').style.color = '#ffee00';
            } else {
                document.getElementById('timer-display').style.color = '#0aff0a';
            }
        }

        // ==========================================
        // LIVE UPDATES
        // ==========================================
        
        function startUpdateLoop() {
            updatePlayers();
            updateInterval = setInterval(updatePlayers, 2000);
        }

        function updatePlayers() {
            const players = TEACHER_SERVER.getPlayers();
            const state = TEACHER_SERVER.getGameState();
            
            const total = players.length;
            const active = players.filter(p => !p.finished && state.status === 'active').length;
            const finished = players.filter(p => p.finished).length;
            const avgScore = total > 0 ? Math.round(players.reduce((a, b) => a + (b.score || 0), 0) / total) : 0;
            
            document.getElementById('total-players').textContent = total;
            document.getElementById('active-players').textContent = active;
            document.getElementById('finished-players').textContent = finished;
            document.getElementById('avg-score').textContent = avgScore;
            
            const tbody = document.getElementById('players-tbody');
            
            if (players.length === 0) {
                tbody.innerHTML = `
                    <tr class="empty-state">
                        <td colspan="6">
                            <div class="empty-icon">👻</div>
                            <div>Все още няма играчи...</div>
                            <div style="font-size: 0.9rem; margin-top: 10px;">Сподели кода с класа!</div>
                        </td>
                    </tr>
                `;
                return;
            }
            
            const sorted = [...players].sort((a, b) => (b.score || 0) - (a.score || 0));
            
            tbody.innerHTML = sorted.map((p, i) => {
                const progress = ((p.level - 1) * 3 + (p.task - 1)) / 9 * 100;
                const status = p.finished ? 'finished' : (state.status === 'active' ? 'playing' : 'waiting');
                const statusText = { waiting: 'Чака', playing: 'Играе', finished: 'Готов' }[status];
                
                return `
                    <tr>
                        <td class="player-rank">${i + 1}</td>
                        <td>
                            <div class="player-name">${p.name || 'Без име'}</div>
                            <div class="player-class">${p.class || ''}</div>

                            </div>
                        </td>
                        <td><span class="status-badge status-${status}">${statusText}</span></td>
                    </tr>
                `;
            }).join('');
        }

                startTimer();

        function generateNewCode() {
            const code = TEACHER_SERVER.generateCode();
            const state = TEACHER_SERVER.getGameState();
            state.classCode = code;
            state.status = 'waiting';

        function startGame() {
            const state = TEACHER_SERVER.getGameState();
            state.status = 'active';
            state.startTime = Date.now();
            endTime = Date.now() + (45 * 60 * 1000);
            state.endTime = endTime;
            TEACHER_SERVER.setGameState(state);
            
            startTimer();
            
            const players = TEACHER_SERVER.getPlayers();
            players.forEach(p => {
                p.finished = false;
                p.score = 0;
                p.level = 1;
                p.task = 1;
                p.accuracy = 0;
            });
            localStorage.setItem('kodat_izhoda_players', JSON.stringify(players));
        }

        function stopGame() {
            const state = TEACHER_SERVER.getGameState();
            state.status = 'stopped';
            state.endTime = Date.now();
            TEACHER_SERVER.setGameState(state);
            
            if (timerInterval) {
                clearInterval(timerInterval);
                timerInterval = null;

            document.getElementById('timer-display').textContent = 'СТОП';
            document.getElementById('timer-display').style.color = '#ff003c';

                timerInterval = null;
